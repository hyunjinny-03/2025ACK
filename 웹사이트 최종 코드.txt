        function normalizeTag(tag) {
            return String(tag || "")
                .trim()
                .replace(/\s+/g, "")          // ê³µë°± ì œê±°
                .replace(/\u00A0/g, "")       // non-breaking space ì œê±°
                .replace(/[\/ï¼]+/g, "/")     // ìŠ¬ë˜ì‹œ/ì „ê°ìŠ¬ë˜ì‹œ í†µì¼
                .replace(/-/g, "")            // í•˜ì´í”ˆ ì œê±°
                .toLowerCase();
        }

        // CSV íƒœê·¸ í•„ë“œ íŒŒì‹±
        function parseTagFieldToArray(raw) {
            if (!raw || typeof raw !== "string") return [];
            const s = raw.trim();

            try {
                return JSON.parse(s.replace(/'/g, '"')).map(x => normalizeTag(x));
            } catch {
                return s
                    .replace(/[\[\]]/g, '')   // ëŒ€ê´„í˜¸ ì œê±°
                    .split(/,|;/)             // ì½¤ë§ˆë‚˜ ì„¸ë¯¸ì½œë¡  ê¸°ì¤€ ë¶„ë¦¬
                    .map(x => normalizeTag(x))
                    .filter(Boolean);
            }
        }
